        public override async Task<AnalysisExecutionResult> ExecuteStageAsync(Analysis analysis)
        {
            AnalysisExecutionResult resultBaseClass = await base.ExecuteStageAsync(analysis); // pega os documentos e embeddings
            
            var fileParts = new List<Part>();
            var allDocuments = resultBaseClass.ReferenceDocuments.Concat(resultBaseClass.AnalysisDocuments);

            //foreach (var document in allDocuments)
            //{

            //    byte[] fileBytes = await _fileStorageService.GetFileAsync(document.FilePath);
            //    var base64Data = Convert.ToBase64String(fileBytes);

            //    fileParts.Add(new Part
            //    {
            //        InlineData = new Blob
            //        {
            //            MimeType = document.FileType.ToMimeType(),
            //            Data = base64Data
            //        }
            //    });

            //}

            foreach (var document in allDocuments)
            {
                byte[] fileBytes = await _fileStorageService.GetFileAsync(document.FilePath);
                var uploadedFileDetails = await _geminiService.UploadFileWithHttpAsync(
                    fileBytes,
                    document.FileType.ToMimeType(),
                    document.FileName
                );

                if (uploadedFileDetails != null)
                {
                    fileParts.Add(new Part
                    {
                        FileData = new FileData
                        {
                            MimeType = uploadedFileDetails.MimeType,
                            FileUri = uploadedFileDetails.Uri
                        }
                    });
                }
                else
                {
                    Console.WriteLine("Falha no upload do arquivo {FileName}. Este arquivo será ignorado na análise.", document.FileName);
                }
            }

            if (fileParts.Any())
            {
                var geminiResponse = await _geminiService.GenerateContentAsync(base.CreateFinalPrompt(resultBaseClass, analysis), fileParts);

                if (geminiResponse.IsSuccess)
                {
                    Console.WriteLine("Resposta da API recebida com sucesso.");
                    Console.WriteLine("------------------ CONTEUDO ------------------");
                    Console.WriteLine(geminiResponse.Content);
                }
                else
                {
                    Console.WriteLine($"Erro da API Gemini: {geminiResponse.ErrorMessage}");
                    //dependendo do erro, tem quer ir trocando as chaves
                }
            }

            return resultBaseClass;
        }
